# This program demonstrates conditional statements in Python
# It shows if, elif, and else usage

# if-else statement
# This program checks whether a person is an adult or a minor

age = int(input("Enter your age: "))

if age >= 18:
    print("You are an adult")
else:
    print("You are a minor")

# if-elif-else statement
# This program takes a user's age as input and determines their age category

age = int(input("Enter your age: "))

if age < 13:
    print("Child")
elif age < 18:
    print("Teenager")
elif age < 60:
    print("Adult")
else:
    print("Senior citizen")
